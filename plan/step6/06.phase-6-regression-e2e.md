### Task 6.1: Playwright 설치 및 설정

**설치:**

**설정 결정 근거:**
- `fullyParallel: false`, `workers: 1`: 테스트가 단일 서버 인스턴스를 공유하므로 PTY 세션 간 간섭 방지
- `webServer`: `dev.js`가 서버(4242)와 프론트엔드(4545)를 동시 기동. `health` 엔드포인트로 서버 준비 확인
- `ignoreHTTPSErrors: true`: 자체 서명 인증서 사용
- 3개 프로젝트(Desktop Chrome, Mobile Safari, Tablet)로 데스크톱/모바일/태블릿 커버

```typescript
import { Page, Locator } from '@playwright/test';

export async function getPaneCount(page: Page): Promise<number> {
  return await page.locator('.pane-leaf').count();
}

export async function getFocusedPane(page: Page): Promise<Locator> {
  return page.locator('.pane-leaf.focused');
}

export async function rightClickPane(page: Page, index: number = 0): Promise<void> {
  const pane = page.locator('.pane-leaf').nth(index);
  await pane.click({ button: 'right' });
  await page.waitForSelector('.context-menu');
}

export async function waitForPaneCount(page: Page, count: number): Promise<void> {
  await page.waitForFunction(
    (expected) => document.querySelectorAll('.pane-leaf').length === expected,
    count,
    { timeout: 10000 }
  );
}

export async function splitPaneViaMenu(
  page: Page,
  direction: 'horizontal' | 'vertical'
): Promise<void> {
  const label = direction === 'vertical' ? '수직 분할' : '수평 분할';
  await page.locator('.pane-leaf').first().click({ button: 'right' });
  await page.waitForSelector('.context-menu');
  await page.locator(`.context-menu-item:has-text("${label}")`).click();
  await page.waitForTimeout(500);
}

export async function closePaneViaMenu(page: Page, paneIndex: number = 0): Promise<void> {
  await page.locator('.pane-leaf').nth(paneIndex).click({ button: 'right' });
  await page.waitForSelector('.context-menu');
  await page.locator('.context-menu-item:has-text("Pane 닫기")').click();
  await page.waitForTimeout(500);
}

export async function waitForTerminal(page: Page): Promise<void> {
  await page.waitForSelector('.xterm-screen', { timeout: 10000 });
}

export async function pressCtrlB(page: Page): Promise<void> {
  await page.keyboard.down('Control');
  await page.keyboard.press('b');
  await page.keyboard.up('Control');
}

export async function swipeLeft(page: Page, selector: string): Promise<void> {
  const el = page.locator(selector);
  const box = await el.boundingBox();
  if (!box) throw new Error('Element not found');
  const startX = box.x + box.width * 0.8;
  const endX = box.x + box.width * 0.2;
  const y = box.y + box.height / 2;
  await page.mouse.move(startX, y);
  await page.mouse.down();
  await page.mouse.move(endX, y, { steps: 10 });
  await page.mouse.up();
}

export async function swipeRight(page: Page, selector: string): Promise<void> {
  const el = page.locator(selector);
  const box = await el.boundingBox();
  if (!box) throw new Error('Element not found');
  const startX = box.x + box.width * 0.2;
  const endX = box.x + box.width * 0.8;
  const y = box.y + box.height / 2;
  await page.mouse.move(startX, y);
  await page.mouse.down();
  await page.mouse.move(endX, y, { steps: 10 });
  await page.mouse.up();
}
```

```typescript
  const item = page.locator(`.context-menu-item:has-text("${label}")`);
  return await item.evaluate(
    (el) => el.classList.contains('disabled') || (el as HTMLButtonElement).disabled
  );
```

```typescript
// tests/e2e/helpers/indexeddb.ts
import { Page } from '@playwright/test';

export async function getIndexedDBLayout(page: Page, sessionId: string): Promise<any> {
  return await page.evaluate(async (sid) => {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open('buildergate', 1);
      request.onsuccess = () => {
        const db = request.result;
        const tx = db.transaction('paneLayouts', 'readonly');
        const store = tx.objectStore('paneLayouts');
        const getReq = store.get(sid);
        getReq.onsuccess = () => resolve(getReq.result);
        getReq.onerror = () => reject(getReq.error);
      };
      request.onerror = () => reject(request.error);
    });
  }, sessionId);
}

export async function clearIndexedDB(page: Page): Promise<void> {
  await page.evaluate(async () => {
    return new Promise<void>((resolve, reject) => {
      const request = indexedDB.deleteDatabase('buildergate');
      request.onsuccess = () => resolve();
      request.onerror = () => reject(request.error);
    });
  });
}

export async function getSavedLayoutCount(page: Page): Promise<number> {
  return await page.evaluate(async () => {
    return new Promise<number>((resolve, reject) => {
      const request = indexedDB.open('buildergate', 1);
      request.onsuccess = () => {
        const db = request.result;
        const tx = db.transaction('savedLayouts', 'readonly');
        const store = tx.objectStore('savedLayouts');
        const countReq = store.count();
        countReq.onsuccess = () => resolve(countReq.result);
        countReq.onerror = () => reject(countReq.error);
      };
      request.onerror = () => reject(request.error);
    });
  });
}
```

### Task 6.2: Pane 분할 E2E 테스트 (`tests/e2e/pane-split.spec.ts`)

TC-6101 ~ TC-6109 커버:

| 테스트 | SRS TC | 검증 내용 |
|--------|--------|----------|
| 수직 분할 | TC-6101 | 좌우 2 Pane 표시, xterm 인스턴스 2개 |
| 8 Pane 제한 | TC-6102 | 분할 메뉴 비활성화 |
| Pane 닫기 | TC-6103 | 나머지 전체 차지, PTY 삭제 |
| 마지막 Pane 닫기 불가 | TC-6104 | 닫기 비활성화 |
| 리사이즈 드래그 | TC-6105 | 비율 변경, xterm 재계산 |
| Ratio 클램핑 | TC-6106 | 0.15 하한 (120px 최소) |
| 줌 토글 | TC-6107 | 단일 표시, [ZOOMED], 복원 |
| Pane 교환 | TC-6108 | 위치 교환, Pane 수 불변 |
| 포커스 이동 | TC-6109 | Ctrl+B 방향키 인접 이동 |

```typescript
import { test, expect } from '@playwright/test';
import { login } from './helpers/auth';
import { getPaneCount, splitPaneViaMenu, closePaneViaMenu, waitForPaneCount } from './helpers/pane';

test.describe('Pane Split System', () => {
  test.beforeEach(async ({ page }) => {
    await login(page);
  });

  test('TC-6101: should split pane vertically creating two side-by-side terminals', async ({ page }) => {
    expect(await getPaneCount(page)).toBe(1);
    await splitPaneViaMenu(page, 'vertical');
    await waitForPaneCount(page, 2);
    const terminals = page.locator('.pane-leaf .xterm');
    expect(await terminals.count()).toBe(2);
  });

  test('TC-6102: should disable split menu at 8 panes', async ({ page }) => {
    for (let i = 0; i < 7; i++) {
      await splitPaneViaMenu(page, 'vertical');
      await waitForPaneCount(page, i + 2);
    }
    expect(await getPaneCount(page)).toBe(8);
    await page.locator('.pane-leaf').first().click({ button: 'right' });
    await page.waitForSelector('.context-menu');
    const hSplit = page.locator('.context-menu-item:has-text("수평 분할")');
    expect(await hSplit.evaluate(el => (el as HTMLButtonElement).disabled)).toBe(true);
    const vSplit = page.locator('.context-menu-item:has-text("수직 분할")');
    expect(await vSplit.evaluate(el => (el as HTMLButtonElement).disabled)).toBe(true);
  });

  test('TC-6103: should close pane and expand remaining', async ({ page }) => {
    await splitPaneViaMenu(page, 'vertical');
    await waitForPaneCount(page, 2);
    await closePaneViaMenu(page, 0);
    await waitForPaneCount(page, 1);
    const pane = page.locator('.pane-leaf').first();
    const box = await pane.boundingBox();
    expect(box!.width).toBeGreaterThan(page.viewportSize()!.width * 0.9);
  });

  test('TC-6104: should disable close for last remaining pane', async ({ page }) => {
    await page.locator('.pane-leaf').first().click({ button: 'right' });
    await page.waitForSelector('.context-menu');
    const closeItem = page.locator('.context-menu-item:has-text("Pane 닫기")');
    expect(await closeItem.evaluate(el => (el as HTMLButtonElement).disabled)).toBe(true);
  });

  test('TC-6105: should resize panes by dragging the boundary', async ({ page }) => {
    await splitPaneViaMenu(page, 'vertical');
    await waitForPaneCount(page, 2);
    const resizer = page.locator('.pane-resizer').first();
    const box = await resizer.boundingBox();
    await resizer.hover();
    await page.mouse.down();
    await page.mouse.move(box!.x + 100, box!.y + box!.height / 2);
    await page.mouse.up();
    const panes = page.locator('.pane-leaf');
    const leftBox = await panes.nth(0).boundingBox();
    const rightBox = await panes.nth(1).boundingBox();
    expect(leftBox!.width).toBeGreaterThan(rightBox!.width);
  });

  test('TC-6106: should clamp resize at minimum ratio (120px)', async ({ page }) => {
    await splitPaneViaMenu(page, 'vertical');
    await waitForPaneCount(page, 2);
    const resizer = page.locator('.pane-resizer').first();
    const box = await resizer.boundingBox();
    await resizer.hover();
    await page.mouse.down();
    await page.mouse.move(50, box!.y + box!.height / 2);
    await page.mouse.up();
    const leftBox = await page.locator('.pane-leaf').nth(0).boundingBox();
    expect(leftBox!.width).toBeGreaterThanOrEqual(120);
  });

  test('TC-6107: should zoom and unzoom a pane', async ({ page }) => {
    await splitPaneViaMenu(page, 'vertical');
    await splitPaneViaMenu(page, 'horizontal');
    await waitForPaneCount(page, 3);
    await page.locator('.pane-leaf').first().click({ button: 'right' });
    await page.locator('.context-menu-item:has-text("줌 토글")').click();
    expect(await page.locator('.pane-leaf:visible').count()).toBe(1);
    await expect(page.locator('text=[ZOOMED]')).toBeVisible();
    await page.locator('.pane-leaf').first().click({ button: 'right' });
    await page.locator('.context-menu-item:has-text("줌 해제")').click();
    expect(await getPaneCount(page)).toBe(3);
    await expect(page.locator('text=[ZOOMED]')).not.toBeVisible();
  });

  test('TC-6108: should swap two panes', async ({ page }) => {
    await splitPaneViaMenu(page, 'vertical');
    await waitForPaneCount(page, 2);
    await page.locator('.pane-leaf').first().click({ button: 'right' });
    await page.locator('.context-menu-item:has-text("Pane 교환")').click();
    await page.locator('.pane-leaf').nth(1).click();
    await page.waitForTimeout(500);
    expect(await getPaneCount(page)).toBe(2);
  });

  test('TC-6109: should move focus with arrow keys in prefix mode', async ({ page }) => {
    await splitPaneViaMenu(page, 'vertical');
    await waitForPaneCount(page, 2);
    await page.locator('.pane-leaf').first().click();
    await page.keyboard.press('Control+b');
    await page.keyboard.press('ArrowRight');
    const rightPane = page.locator('.pane-leaf').nth(1);
    await expect(rightPane).toHaveClass(/focused/);
  });
});
```

```typescript
import { test, expect } from '@playwright/test';
import { login } from './helpers/auth';
import { splitPaneViaMenu, waitForPaneCount } from './helpers/pane';
import { openContextMenu, hoverSubmenu } from './helpers/menu';

test.describe('Context Menus', () => {
  test.beforeEach(async ({ page }) => { await login(page); });

  test('TC-6201: should show context menu on terminal right-click', async ({ page }) => {
    const pane = page.locator('.pane-leaf').first();
    await openContextMenu(page, pane);
    await expect(page.locator('.context-menu')).toBeVisible();
    await expect(page.locator('.context-menu-item:has-text("수평 분할")')).toBeVisible();
    await expect(page.locator('.context-menu-item:has-text("수직 분할")')).toBeVisible();
    await expect(page.locator('.context-menu-item:has-text("줌 토글")')).toBeVisible();
    await expect(page.locator('.context-menu-item:has-text("Pane 닫기")')).toBeVisible();
  });

  test('TC-6202: should equalize panes from resizer menu', async ({ page }) => {
    await splitPaneViaMenu(page, 'vertical');
    await waitForPaneCount(page, 2);
    const resizer = page.locator('.pane-resizer').first();
    await openContextMenu(page, resizer);
    await expect(page.locator('.context-menu-item:has-text("균등 분할")')).toBeVisible();
    await expect(page.locator('.context-menu-item:has-text("방향 전환")')).toBeVisible();
    await page.locator('.context-menu-item:has-text("균등 분할")').click();
    await page.waitForTimeout(300);
    const panes = page.locator('.pane-leaf');
    const leftBox = await panes.nth(0).boundingBox();
    const rightBox = await panes.nth(1).boundingBox();
    expect(Math.abs(leftBox!.width - rightBox!.width)).toBeLessThan(10);
  });

  test('TC-6203: should apply quad preset from TabBar menu', async ({ page }) => {
    const tab = page.locator('.tab-item').first();
    await openContextMenu(page, tab);
    await hoverSubmenu(page, '프리셋 레이아웃');
    await page.locator('.context-menu-submenu .context-menu-item:has-text("4분할")').click();
    const confirm = page.locator('button:has-text("확인")');
    if (await confirm.isVisible()) await confirm.click();
    await waitForPaneCount(page, 4);
  });

  test('TC-6204: should show submenu with arrow on hover', async ({ page }) => {
    const tab = page.locator('.tab-item').first();
    await openContextMenu(page, tab);
    const item = page.locator('.context-menu-item.has-children:has-text("프리셋 레이아웃")');
    await expect(item).toBeVisible();
    await item.hover();
    await page.waitForSelector('.context-menu-submenu', { state: 'visible' });
    await expect(page.locator('.context-menu-submenu .context-menu-item:has-text("단일")')).toBeVisible();
    await expect(page.locator('.context-menu-submenu .context-menu-item:has-text("좌우 분할")')).toBeVisible();
  });

  test('TC-6205: should show bottom sheet on mobile long-press', async ({ page }) => {
    await page.setViewportSize({ width: 390, height: 844 });
    const pane = page.locator('.pane-leaf').first();
    const box = await pane.boundingBox();
    const x = box!.x + box!.width / 2;
    const y = box!.y + box!.height / 2;
    await page.evaluate(async ({ x, y }) => {
      const el = document.elementFromPoint(x, y)!;
      const touch = new Touch({ identifier: 0, target: el, clientX: x, clientY: y });
      el.dispatchEvent(new TouchEvent('touchstart', { touches: [touch], bubbles: true }));
      await new Promise(r => setTimeout(r, 550));
      el.dispatchEvent(new TouchEvent('touchend', { bubbles: true }));
    }, { x, y });
    await expect(page.locator('.bottom-sheet-menu')).toBeVisible({ timeout: 2000 });
  });
});
```

```typescript
import { test, expect } from '@playwright/test';
import { login } from './helpers/auth';
import { splitPaneViaMenu, waitForPaneCount, getPaneCount } from './helpers/pane';

test.describe('Performance Validation', () => {
  test.beforeEach(async ({ page }) => { await login(page); });

  test('TC-NFR-01: 8 panes 60fps during resize', async ({ page }) => {
    for (let i = 0; i < 7; i++) {
      await splitPaneViaMenu(page, i % 2 === 0 ? 'vertical' : 'horizontal');
      await page.waitForTimeout(500);
    }
    await waitForPaneCount(page, 8);
    const client = await page.context().newCDPSession(page);
    await client.send('Performance.enable');
    const resizer = page.locator('.pane-resizer').first();
    if (await resizer.isVisible()) {
      const box = await resizer.boundingBox();
      if (box) {
        await page.mouse.move(box.x + box.width / 2, box.y + box.height / 2);
        await page.mouse.down();
        for (let i = 0; i < 10; i++) {
          await page.mouse.move(box.x + 50 + i * 5, box.y + box.height / 2);
          await page.waitForTimeout(16);
        }
        await page.mouse.up();
      }
    }
    const metrics = await client.send('Performance.getMetrics');
    const ld = metrics.metrics.find((m: any) => m.name === 'LayoutDuration');
    if (ld) expect(ld.value).toBeLessThan(0.1);
    await client.send('Performance.disable');
  });

  test('TC-NFR-02: memory under 500MB with 8 panes', async ({ page }) => {
    for (let i = 0; i < 7; i++) {
      await splitPaneViaMenu(page, 'vertical');
      await page.waitForTimeout(500);
    }
    await waitForPaneCount(page, 8);
    await page.waitForTimeout(2000);
    const mem = await page.evaluate(() => {
      if ('memory' in performance) {
        return (performance as any).memory.usedJSHeapSize;
      }
      return null;
    });
    if (mem) {
      const mb = mem / (1024 * 1024);
      console.log(`JS Heap: ${mb.toFixed(1)} MB`);
      expect(mb).toBeLessThan(500);
    }
  });

  test('TC-NFR-03: 360px mobile carousel with 8 panes', async ({ page }) => {
    await page.setViewportSize({ width: 1920, height: 1080 });
    for (let i = 0; i < 7; i++) {
      await splitPaneViaMenu(page, 'vertical');
      await page.waitForTimeout(300);
    }
    await waitForPaneCount(page, 8);
    await page.setViewportSize({ width: 360, height: 640 });
    await page.waitForTimeout(500);
    await expect(page.locator('.pane-carousel')).toBeVisible();
    const dots = page.locator('.pane-dot');
    expect(await dots.count()).toBe(8);
    for (let i = 0; i < 8; i++) {
      const box = await dots.nth(i).boundingBox();
      expect(box!.x).toBeGreaterThanOrEqual(0);
      expect(box!.x + box!.width).toBeLessThanOrEqual(360);
    }
  });

  test('TC-NFR-04: full pane operation via keyboard only', async ({ page }) => {
    await page.locator('.pane-leaf').first().click();
    await page.keyboard.press('Control+b');
    await page.keyboard.press('Shift+5');
    await waitForPaneCount(page, 2);
    await page.keyboard.press('Control+b');
    await page.keyboard.press('ArrowRight');
    await page.keyboard.press('Control+b');
    await page.keyboard.press('Shift+Quote');
    await waitForPaneCount(page, 3);
    await page.keyboard.press('Control+b');
    await page.keyboard.press('q');
    await expect(page.locator('.pane-number-overlay').first()).toBeVisible();
    await page.keyboard.press('0');
    await expect(page.locator('.pane-number-overlay').first()).not.toBeVisible();
    await page.keyboard.press('Control+b');
    await page.keyboard.press('z');
    await expect(page.locator('.zoom-indicator')).toBeVisible();
    await page.keyboard.press('Control+b');
```
