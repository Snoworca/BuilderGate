- Phase 1~5 모두 완료
- 서버: https://localhost:4242 (HTTPS)
- 프론트엔드: http://localhost:4545 (Vite dev proxy)

### Task 6.2: 테스트 헬퍼 (`tests/e2e/helpers.ts`)

```typescript
import { Page, expect } from '@playwright/test';

export async function login(page: Page) {
  await page.goto('/');
  await page.fill('input[type="password"]', 'admin');  // 기본 패스워드
  await page.click('button[type="submit"]');
  await page.waitForSelector('.workspace-screen');
}

export async function waitForTerminal(page: Page) {
  await page.waitForSelector('.xterm-screen', { timeout: 10000 });
}

export async function rightClickPane(page: Page, index: number) {
  const pane = page.locator('.pane-leaf').nth(index);
  await pane.click({ button: 'right' });
  await page.waitForSelector('.context-menu');
}

export async function selectMenuItem(page: Page, label: string) {
  await page.click(`.context-menu-item:has-text("${label}")`);
}

export async function getPaneCount(page: Page): Promise<number> {
  return page.locator('.pane-leaf').count();
}

export async function clearIndexedDB(page: Page) {
  await page.evaluate(() => {
    return new Promise<void>((resolve) => {
      const req = indexedDB.deleteDatabase('buildergate');
      req.onsuccess = () => resolve();
      req.onerror = () => resolve();
    });
  });
}

export async function swipeLeft(page: Page, selector: string) {
  const el = page.locator(selector);
  const box = await el.boundingBox();
  if (!box) throw new Error('Element not found');
  const startX = box.x + box.width * 0.8;
  const endX = box.x + box.width * 0.2;
  const y = box.y + box.height / 2;
  await page.touchscreen.tap(startX, y);
  // Simulate swipe via mouse (Playwright touchscreen limitation)
  await page.mouse.move(startX, y);
  await page.mouse.down();
  await page.mouse.move(endX, y, { steps: 10 });
  await page.mouse.up();
}

export async function pressCtrlB(page: Page) {
  await page.keyboard.down('Control');
  await page.keyboard.press('b');
  await page.keyboard.up('Control');
}
```

### Task 6.9: 성능 검증 (`tests/e2e/performance.spec.ts`)

| 테스트 | SRS NFR | 검증 방법 |
|--------|---------|----------|
| 8 Pane 60fps | NFR-6101 | CDP `Performance.enable`, FPS 측정 |
| 메모리 500MB | NFR-6106 | `performance.memory.usedJSHeapSize` |
| 360px 모바일 | NFR-6107 | Mobile 프로젝트 + 8 Pane |
| 키보드 접근성 | NFR-6108 | Tab + Ctrl+B로 모든 조작 |
| 동시 고출력 | NFR-6111 | 8 Pane 동시 `yes` 명령 실행 |

```typescript
test('8 panes under 500MB memory', async ({ page }) => {
  await login(page);
  // Create 8 panes
  for (let i = 0; i < 7; i++) {
    await rightClickPane(page, 0);
    await selectMenuItem(page, '수직 분할');
    await page.waitForTimeout(1000);
  }
  expect(await getPaneCount(page)).toBe(8);

  const memory = await page.evaluate(() =>
    (performance as any).memory?.usedJSHeapSize
  );
  if (memory) {
    expect(memory).toBeLessThan(500 * 1024 * 1024);
  }
});
```

## 실행 방법

```bash
# 전체 테스트 (headless)
cd frontend && npx playwright test

# 특정 스위트
npx playwright test pane-split

# 특정 프로젝트
npx playwright test --project="Mobile Safari"
npx playwright test --project="Desktop Chrome"

# UI 모드 (디버깅)
npx playwright test --ui

# 실패 트레이스 확인
npx playwright show-trace test-results/*/trace.zip

# JUnit 리포트 (CI용)
npx playwright test --reporter=junit --output-file=test-results.xml

# HTML 리포트
npx playwright test --reporter=html
npx playwright show-report
```
