## 1. 목표 (Objective)

Mdir(1991, 최정한) 스타일의 파일 관리자를 구현한다. DOS 시대의 미니멀 흑백 UI를 웹으로 재현하며, 2-컬럼 파일 리스팅, 키보드 네비게이션, 파일 조작(Copy/Move/Delete) 기능을 포함한다.

### 3.2 데이터 흐름

```
[TabBar] ──click "Files"──> [MdirPanel] ──mount──> [useFileBrowser]
                                                        │
                                                        ├── GET /api/sessions/:id/cwd
                                                        │     → FileService.getCwd()
                                                        │     → /proc/<pid>/cwd (Linux) 또는 PowerShell (Win)
                                                        │
                                                        ├── GET /api/sessions/:id/files?path=...
                                                        │     → FileService.listDirectory()
                                                        │     → fs.readdir() + fs.stat()
                                                        │     → pathValidator.validatePath()
                                                        │
                                                        ├── POST /api/sessions/:id/files/copy
                                                        │     → FileService.copyFile()
                                                        │     → fs.copyFile() (+ recursive for dirs)
                                                        │
                                                        ├── POST /api/sessions/:id/files/move
                                                        │     → FileService.moveFile()
                                                        │     → fs.rename()
                                                        │
                                                        └── DELETE /api/sessions/:id/files?path=...
                                                              → FileService.deleteFile()
                                                              → fs.unlink() / fs.rm({ recursive: true })
```

// frontend/src/__tests__/hooks/useKeyboardNav.test.ts
import { describe, it, expect } from 'vitest';
import { renderHook, act } from '@testing-library/react';
import { useKeyboardNav } from '../../hooks/useKeyboardNav';

const entries = Array.from({ length: 10 }, (_, i) => ({
  name: `file${i}`,
  type: 'file' as const,
  size: 1024,
  modified: '2026-02-15T00:00:00Z',
}));

describe('useKeyboardNav', () => {
  it('should navigate down within column', () => {
    const { result } = renderHook(() =>
      useKeyboardNav(entries, 2, { onEnter: vi.fn(), onEsc: vi.fn() })
    );

    // Initial: index 0 (row 0, col 0)
    expect(result.current.selectedIndex).toBe(0);

    // ArrowDown: row 0 → row 1 (still col 0)
    act(() => {
      result.current.handleKeyDown(new KeyboardEvent('keydown', { key: 'ArrowDown' }));
    });
    expect(result.current.selectedIndex).toBe(1);
  });

  it('should navigate between columns with left/right', () => {
    const { result } = renderHook(() =>
      useKeyboardNav(entries, 2, { onEnter: vi.fn(), onEsc: vi.fn() })
    );

    // rowCount = ceil(10/2) = 5
    // Move to row 1 first
    act(() => {
      result.current.handleKeyDown(new KeyboardEvent('keydown', { key: 'ArrowDown' }));
    });
    expect(result.current.selectedIndex).toBe(1); // row 1, col 0

    // ArrowRight: col 0 → col 1, same row
    act(() => {
      result.current.handleKeyDown(new KeyboardEvent('keydown', { key: 'ArrowRight' }));
    });
    expect(result.current.selectedIndex).toBe(6); // row 1, col 1 = 5 + 1
  });

  it('should clamp at boundaries', () => {
    const { result } = renderHook(() =>
      useKeyboardNav(entries, 2, { onEnter: vi.fn(), onEsc: vi.fn() })
    );

    // At index 0, ArrowUp should stay at 0
    act(() => {
      result.current.handleKeyDown(new KeyboardEvent('keydown', { key: 'ArrowUp' }));
    });
    expect(result.current.selectedIndex).toBe(0);

    // At index 0, ArrowLeft should stay at 0
    act(() => {
      result.current.handleKeyDown(new KeyboardEvent('keydown', { key: 'ArrowLeft' }));
    });
    expect(result.current.selectedIndex).toBe(0);
  });

  it('should handle Home and End', () => {
    const { result } = renderHook(() =>
      useKeyboardNav(entries, 2, { onEnter: vi.fn(), onEsc: vi.fn() })
    );

    act(() => {
      result.current.handleKeyDown(new KeyboardEvent('keydown', { key: 'End' }));
    });
    expect(result.current.selectedIndex).toBe(9);

    act(() => {
      result.current.handleKeyDown(new KeyboardEvent('keydown', { key: 'Home' }));
    });
    expect(result.current.selectedIndex).toBe(0);
  });
});
