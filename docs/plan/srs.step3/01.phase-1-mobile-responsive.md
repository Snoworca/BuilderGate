#### 📁 **Step 1: Create `useResponsive` Hook** (🔴 High Priority)

**파일**: `frontend/src/hooks/useResponsive.ts` (NEW)

**목적**: Breakpoint 감지 및 sidebar 상태 관리

**매핑**: FR-1801 (Responsive Layout Breakpoints)

**복잡도**: ⭐⭐ (Medium)

**구현 상세:**

```typescript
/**
 * useResponsive.ts
 *
 * Detects viewport breakpoint and manages sidebar visibility state.
 *
 * Breakpoint: 768px
 * - Mobile: < 768px (sidebar overlay, initially hidden)
 * - Desktop: >= 768px (sidebar always visible, cannot be toggled)
 *
 * Returns:
 * - isMobile: boolean
 * - sidebarOpen: boolean (only meaningful on mobile)
 * - toggleSidebar: () => void
 * - closeSidebar: () => void (for overlay click)
 */

import { useState, useEffect, useCallback } from 'react';

const MOBILE_BREAKPOINT = 768; // px

interface UseResponsiveReturn {
  isMobile: boolean;
  sidebarOpen: boolean;
  toggleSidebar: () => void;
  closeSidebar: () => void;
}

export const useResponsive = (): UseResponsiveReturn => {
  const [isMobile, setIsMobile] = useState<boolean>(
    window.innerWidth < MOBILE_BREAKPOINT
  );

  const [sidebarOpen, setSidebarOpen] = useState<boolean>(false);

  // Detect breakpoint changes
  useEffect(() => {
    const handleResize = () => {
      const mobile = window.innerWidth < MOBILE_BREAKPOINT;
      setIsMobile(mobile);

      // Auto-close sidebar when switching to desktop
      if (!mobile) {
        setSidebarOpen(false);
      }
    };

    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  const toggleSidebar = useCallback(() => {
    if (isMobile) {
      setSidebarOpen(prev => !prev);
    }
  }, [isMobile]);

  const closeSidebar = useCallback(() => {
    setSidebarOpen(false);
  }, []);

  return {
    isMobile,
    sidebarOpen,
    toggleSidebar,
    closeSidebar,
  };
};
```

**테스트 포인트:**
- [ ] `window.innerWidth = 767` → `isMobile = true`
- [ ] `window.innerWidth = 768` → `isMobile = false`
- [ ] Resize 이벤트 시 `isMobile` 자동 업데이트
- [ ] Desktop 전환 시 `sidebarOpen` 자동 false

```typescript
// TerminalView.tsx (enhanced)
useEffect(() => {
  const term = new Terminal({ /* options */ });
  const fitAddon = new FitAddon();
  term.loadAddon(fitAddon);
  term.open(terminalRef.current!);

  fitAddon.fit(); // Initial fit

  // Handle window resize (debounced for performance)
  let resizeTimer: number;
  const handleResize = () => {
    clearTimeout(resizeTimer);
    resizeTimer = window.setTimeout(() => {
      try {
        fitAddon.fit();

        // Send resize event to backend (PTY resize)
        if (sessionId) {
          fetch(`/api/sessions/${sessionId}/resize`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              cols: term.cols,
              rows: term.rows,
            }),
          });
        }
      } catch (error) {
        console.error('Terminal resize error:', error);
      }
    }, 100); // 100ms debounce
  };

  window.addEventListener('resize', handleResize);

  return () => {
    window.removeEventListener('resize', handleResize);
    clearTimeout(resizeTimer);
    term.dispose();
  };
}, [sessionId]);
```

// useResponsive.test.ts
import { renderHook, act } from '@testing-library/react';
import { useResponsive } from './useResponsive';

describe('useResponsive', () => {
  beforeEach(() => {
