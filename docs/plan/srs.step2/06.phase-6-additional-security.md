  ```typescript
  export function createCorsMiddleware(config: CORSConfig) {
    return cors({
      origin: (origin, callback) => {
        if (!config.allowedOrigins.length) {
          // Empty = same-origin only
          if (!origin) return callback(null, true);
          return callback(new Error('Not allowed by CORS'));
        }
        if (config.allowedOrigins.includes(origin)) {
          return callback(null, true);
        }
        return callback(new Error('Not allowed by CORS'));
      },
      credentials: config.allowCredentials,
      maxAge: config.maxAge,
      methods: config.allowedMethods,
      allowedHeaders: config.allowedHeaders
    });
  }
  ```

  ```typescript
  export function validateBody<T>(schema: ZodSchema<T>) {
    return (req: Request, res: Response, next: NextFunction) => {
      const result = schema.safeParse(req.body);

      if (!result.success) {
        const errors = result.error.errors.map(e => ({
          field: e.path.join('.'),
          message: e.message
        }));

        return res.status(400).json({
          error: {
            code: 'INVALID_REQUEST',
            message: 'Validation failed',
            details: { errors }
          }
        });
      }

      req.body = result.data;
      next();
    };
  }
  ```

### 3.4 Environment Variable Filtering

- [ ] **3.4.1** Create `server/src/services/EnvFilterService.ts`
  ```typescript

    private config: EnvFilterConfig;

    constructor(config: EnvFilterConfig)
    filterEnvironment(env: NodeJS.ProcessEnv): NodeJS.ProcessEnv
    isAllowed(key: string): boolean
    private matchesPattern(key: string, pattern: string): boolean
    private matchesPrefix(key: string, prefix: string): boolean
  }
  ```

- [ ] **3.4.2** Implement filtering logic
  ```typescript
  filterEnvironment(env: NodeJS.ProcessEnv): NodeJS.ProcessEnv {
    const filtered: NodeJS.ProcessEnv = {};

    for (const [key, value] of Object.entries(env)) {
      if (this.isAllowed(key)) {
        filtered[key] = value;
      }
    }

    return filtered;
  }

  isAllowed(key: string): boolean {
    // 1. allowList에 있으면 허용
    if (this.config.allowList.includes(key)) return true;

    // 2. blockPatterns 매칭 시 차단
    for (const pattern of this.config.blockPatterns) {
      if (this.matchesPattern(key, pattern)) return false;
    }

    // 3. blockPrefixes 매칭 시 차단
    for (const prefix of this.config.blockPrefixes) {
      if (key.startsWith(prefix)) return false;
    }

    // 4. 나머지 허용
    return true;
  }
  ```

- [ ] **3.4.3** Implement pattern matching
  ```typescript
  private matchesPattern(key: string, pattern: string): boolean {
    // Convert glob pattern to regex
    // *_SECRET → /.*_SECRET$/
    // *_KEY → /.*_KEY$/
    const regex = pattern
      .replace(/\*/g, '.*')
      .replace(/\?/g, '.');
    return new RegExp(`^${regex}$`).test(key);
  }
  ```

- [ ] **3.5.2** Create `server/src/services/AuditService.ts`
  ```typescript
  import winston from 'winston';
  import DailyRotateFile from 'winston-daily-rotate-file';

  class AuditService {
    private logger: winston.Logger;
    private config: LoggingConfig;

    constructor(config: LoggingConfig) {
      this.config = config;
      this.logger = this.createLogger();
    }

    logSecurityEvent(event: SecurityEvent): void {
      const entry: AuditEntry = {
        timestamp: new Date().toISOString(),
        level: event.level,
        event: event.type,
        ip: event.ip,
        userAgent: event.userAgent,
        details: this.sanitizeDetails(event.details)
      };

      this.logger.log(event.level.toLowerCase(), entry);
    }

    logAuthEvent(type: AuthEventType, details: AuthEventDetails): void {
      this.logSecurityEvent({
        level: type.startsWith('LOGIN_FAILED') ? 'WARN' : 'INFO',
        type,
        ip: details.ip,
        userAgent: details.userAgent,
        details
      });
    }

    logSessionEvent(type: SessionEventType, details: SessionDetails): void {
      this.logSecurityEvent({
        level: 'INFO',
        type,
        ip: details.ip,
        userAgent: details.userAgent || '',
        details
      });
    }

    private sanitizeDetails(details: Record<string, unknown>): Record<string, unknown> {
      // Remove sensitive fields
      const { password, token, otp, ...safe } = details;
      return safe;
    }
  }
  ```

- [ ] **3.5.3** Configure winston logger
  ```typescript
  private createLogger(): winston.Logger {
    const transports: winston.transport[] = [
      new winston.transports.Console({
        format: winston.format.combine(
          winston.format.timestamp(),
          winston.format.json()
        )
      })
    ];

    if (this.config.securityLog) {
      transports.push(
        new DailyRotateFile({
          filename: `${this.config.logDir}/security-%DATE%.log`,
          datePattern: 'YYYY-MM-DD',
          maxSize: this.config.maxFileSize,
          maxFiles: this.config.maxFiles,
          compress: this.config.compress,
          format: winston.format.combine(
            winston.format.timestamp(),
            winston.format.json()
          )
        })
      );
    }

    return winston.createLogger({
      level: this.config.level,
      transports
    });
  }
  ```

### 3.6 Audit Logger Middleware

- [ ] **3.6.1** Create `server/src/middleware/auditLogger.ts`
  ```typescript
  export function createAuditLoggerMiddleware(auditService: AuditService) {
    return (req: Request, res: Response, next: NextFunction) => {
      const startTime = Date.now();

      res.on('finish', () => {
        const duration = Date.now() - startTime;

        // Log security-relevant requests only
        if (isSecurityRelevant(req.path)) {
          auditService.logSecurityEvent({
            level: res.statusCode >= 400 ? 'WARN' : 'INFO',
            type: getEventType(req, res),
            ip: getClientIP(req),
            userAgent: req.headers['user-agent'] || '',
            details: {
              method: req.method,
              path: req.path,
              statusCode: res.statusCode,
              duration
            }
          });
        }
      });

      next();
    };
  }

  function isSecurityRelevant(path: string): boolean {
    return path.includes('/auth/') ||
           path.includes('/sessions');
  }
  ```

## 4. Detailed Specifications

### 4.1 CORS Configuration (FR-1201)

```json5
{
  security: {
```

### 4.2 Validation Rules (FR-1501)

| Field | Type | Constraints | Error Code |
|-------|------|-------------|------------|
| password | string | 1-128자, ASCII 32-126 | INVALID_PASSWORD_FORMAT |
| otpCode | string | ^[0-9]{4,8}$ | INVALID_OTP_FORMAT |
| sessionName | string | 1-100자, ^[a-zA-Z0-9\s\-_]+$ | INVALID_SESSION_NAME |
| cols | integer | 10-500 | INVALID_TERMINAL_SIZE |
| rows | integer | 5-200 | INVALID_TERMINAL_SIZE |
| tempToken | string | UUID v4 | INVALID_TEMP_TOKEN |

### 4.3 Environment Filter (FR-1701)

```json5
{
  security: {
    envFilter: {
      blockPatterns: ["*_SECRET", "*_KEY", "*_PASSWORD", "*_TOKEN",
                      "*_CREDENTIALS", "*_API_KEY"],
      blockPrefixes: ["AWS_", "AZURE_", "GCP_", "GOOGLE_"],
      allowList: ["PATH", "HOME", "TERM", "LANG", "USER", "SHELL"]

  }
}
```

## 7. Acceptance Criteria

### 7.1 FR-1201: Strict CORS Policy
- [ ] AC-1201-1: 허용된 origin → Access-Control-Allow-Origin 포함
- [ ] AC-1201-2: 허용되지 않은 origin → CORS 헤더 없음
- [ ] AC-1201-3: Preflight 요청 정상 처리

### 7.2 FR-1501: Request Validation
- [ ] AC-1501-1: 유효하지 않은 입력 → 400 Bad Request
- [ ] AC-1501-2: 에러 응답에 field명과 이유 포함

### 7.3 FR-1502: Command Injection Prevention
- [ ] AC-1502-1: 세션 이름에 쉘 메타문자 포함 시 제거됨
- [ ] AC-1502-2: 로그 메시지에 JSON injection 불가

### 7.4 FR-1701: Filtered Environment
- [ ] AC-1701-1: AWS_SECRET_ACCESS_KEY → PTY에 전달 안됨
- [ ] AC-1701-2: PATH → PTY에 전달됨
- [ ] AC-1701-3: CUSTOM_API_KEY → PTY에 전달 안됨

### 7.5 FR-1401: Security Event Logging
- [ ] AC-1401-1: 모든 보안 이벤트가 로깅됨
- [ ] AC-1401-2: 로그에 민감 정보(비밀번호, OTP) 미포함
- [ ] AC-1401-3: 타임스탬프 ISO 8601 형식

### 7.6 FR-1402: Log Rotation
- [ ] AC-1402-1: 10MB 초과 시 새 파일 생성
- [ ] AC-1402-2: 31번째 파일 생성 시 가장 오래된 파일 삭제
