  isAccountLocked(ip: string): LockStatus {
    const data = this.loginAttempts.get(ip);
    if (!data || !data.lockedUntil) return { locked: false };

    if (Date.now() < data.lockedUntil) {
      return {
        locked: true,
        lockedUntil: new Date(data.lockedUntil).toISOString(),
        remainingMs: data.lockedUntil - Date.now()
      };
    }

    // 잠금 해제
    data.lockedUntil = null;
    data.attempts = 0;
    return { locked: false };
  }
  ```

  getLockoutDuration(lockCount: number): number {
    const base = this.config.lockoutDurationMs;
    const multiplier = this.config.lockoutMultiplier;
    const duration = base * Math.pow(multiplier, lockCount);
    return Math.min(duration, 86400000); // 최대 24시간
  }
  ```

### 3.3 Progressive Delay

- [ ] **3.3.1** Implement delay calculation
  ```typescript
  getProgressiveDelay(attempts: number): number {
    // delay = min(200 * 2^(attempts-1), 10000)
    if (attempts <= 0) return 0;
    const delay = 200 * Math.pow(2, attempts - 1);
    return Math.min(delay, 10000);
  }
  ```

### 3.4 Rate Limiting Service

- [ ] **3.4.1** Create `server/src/services/RateLimitService.ts`
  ```typescript
  class RateLimitService {
    private requestCounts: Map<string, RequestCount[]>;
    private ipBlacklist: Map<string, number>;
    private lockoutCounts: Map<string, number>;

    constructor(config: RateLimitConfig)
    checkLimit(ip: string, endpoint: string): RateLimitResult
    trackLockout(ip: string): void
    isBlacklisted(ip: string): BlacklistStatus
    addToBlacklist(ip: string, durationMs: number): void
    getRateLimitHeaders(ip: string): RateLimitHeaders
  }
  ```

- [ ] **3.4.2** Implement sliding window rate limiting
  ```typescript
  checkLimit(ip: string, endpoint: string): RateLimitResult {
    const now = Date.now();
    const windowStart = now - this.config.windowMs;

    // 윈도우 내 요청 카운트
    const requests = this.requestCounts.get(ip) || [];
    const recentRequests = requests.filter(r => r.timestamp > windowStart);

    // 엔드포인트별 제한 확인
    const limit = endpoint === 'login'
      ? this.config.loginMaxRequests
      : this.config.maxRequests;

    if (recentRequests.length >= limit) {
      return {
        allowed: false,
        retryAfter: windowStart + this.config.windowMs - now
      };
    }

    // 요청 기록
    recentRequests.push({ timestamp: now, endpoint });
    this.requestCounts.set(ip, recentRequests);

    return {
      allowed: true,
      remaining: limit - recentRequests.length,
      reset: windowStart + this.config.windowMs
    };
  }
  ```

### 3.5 Rate Limiter Middleware

- [ ] **3.5.1** Create `server/src/middleware/rateLimiter.ts`
  ```typescript
  export function createRateLimiterMiddleware(
    rateLimitService: RateLimitService
  ) {
    return (req: Request, res: Response, next: NextFunction) => {
      const ip = getClientIP(req);

      // 블랙리스트 확인
      const blacklistStatus = rateLimitService.isBlacklisted(ip);
      if (blacklistStatus.blocked) {
        return res.status(403).json({
          error: {
            code: 'IP_BLACKLISTED',
            message: 'Your IP has been blocked',
            blockedUntil: blacklistStatus.blockedUntil
          }
        });
      }

      // Rate limit 확인
      const endpoint = req.path.includes('/auth/login') ? 'login' : 'general';
      const result = rateLimitService.checkLimit(ip, endpoint);

      // 헤더 설정
      const headers = rateLimitService.getRateLimitHeaders(ip);
      res.setHeader('X-RateLimit-Limit', headers.limit);
      res.setHeader('X-RateLimit-Remaining', headers.remaining);
      res.setHeader('X-RateLimit-Reset', headers.reset);

      if (!result.allowed) {
        res.setHeader('Retry-After', Math.ceil(result.retryAfter / 1000));
        return res.status(429).json({
          error: {
            code: 'RATE_LIMITED',
            message: 'Too many requests',
            retryAfter: result.retryAfter
          }
        });
      }

      next();
    };
  }
  ```

- [ ] **3.5.2** Implement IP extraction
  ```typescript
  function getClientIP(req: Request): string {
    const forwarded = req.headers['x-forwarded-for'];
    if (forwarded) {
      return (typeof forwarded === 'string' ? forwarded : forwarded[0])
        .split(',')[0].trim();
    }
    return req.ip || req.socket.remoteAddress || 'unknown';
  }
  ```

    const count = (this.lockoutCounts.get(ip) || 0) + 1;
    this.lockoutCounts.set(ip, count);

    if (count >= this.config.autoBlacklistThreshold) {

- [ ] **3.6.3** Implement manual blacklist check
  ```typescript
  isBlacklisted(ip: string): BlacklistStatus {
    // 수동 블랙리스트 확인
    if (this.config.ipBlacklist.includes(ip)) {
      return { blocked: true, permanent: true };
    }

    // 자동 블랙리스트 확인
    const expiresAt = this.ipBlacklist.get(ip);
    if (expiresAt) {
      if (Date.now() < expiresAt) {
        return {
          blocked: true,
          blockedUntil: new Date(expiresAt).toISOString()
        };
      }
      this.ipBlacklist.delete(ip);
    }

    return { blocked: false };
  }
  ```
